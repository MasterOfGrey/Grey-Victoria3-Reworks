# Overwrites - these will be part of the next morgenrote update
curtiss_has_aeroplane_production_size_x_trigger = {
	# LEVEL = level of the Building
	any_scope_building = {
		# Checks for ANY PM's that produce aeroplanes, and covers if a mod has separated planes off into their own industry.
		modifier:goods_output_aeroplanes_add > 0
		level >= $LEVEL$
	}
}
curtiss_has_aeroplane_production_size_x_tooltip_trigger = {
	# LEVEL = level of the Building
	save_temporary_scope_value_as = { #This is used to display the level in the tooltip
		name = building_level
		value = $LEVEL$
	}
	custom_tooltip = {
		text = curtiss_has_aeroplane_production_size_x_tooltip_trigger_tt
		any_scope_building = {
			# Checks for ANY PM's that produce aeroplanes, and covers if a mod has separated planes off into their own industry.
			modifier:goods_output_aeroplanes_add > 0
			level >= $LEVEL$
		}
	}
}
