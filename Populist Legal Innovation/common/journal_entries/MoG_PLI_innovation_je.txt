je_mog_pli_innovation = {
	icon = "gfx/interface/icons/event_icons/event_industry.dds"
	group = je_group_internal_affairs
	should_be_pinned_by_default = yes
	can_revolution_inherit = yes
	weight = 9000
	is_shown_when_inactive = { NOT = { is_country_type = decentralized } }
	possible = {
		has_journal_entry = je_mog_pli_investment
		is_revolutionary = no
	}
	
	scripted_progress_bar = mog_pli_innovation_tracking_bar
	scripted_progress_bar = mog_pli_proliferation_tracking_bar
	
	immediate = {
		trigger_event = { id = PLI_secret_ev.2 days = 1 }	# Set bar starting level 1 day later, effect doesn't work in immediate
		set_variable = {									# Set the variables now even though the event changes it straight after
			name = mog_pli_innovation_current_value			# This stops it from giving errors if the je initialises on the weekly pulse
			value = "je:je_mog_pli_innovation.scripted_bar_progress(mog_pli_innovation_tracking_bar)"
		}
		set_variable = {
			name = mog_pli_proliferation_current_value
			value = "je:je_mog_pli_innovation.scripted_bar_progress(mog_pli_proliferation_tracking_bar)"
		}
		create_com_progress_bar_with_both = {
			progress_bar = mog_pli_innovation_tracking_bar #just a unique ID, any loc string is fine
			index = 0 #0-indexed position in JE
			color = default #just copying base color
			bar_increase_color = green #drift color when increasing, any bar color is valid, including white
			bar_decrease_color = bad #drift color when decreasing, any bar color is valid
			drift_value = mog_pli_innovation_tracking_bar_weekly_drift #script value that sets drift amount
			target_type = gold_bar #target marker, gold_bar, gold_marker, journal_icon, and "any bar color"_line are options
			target_value = mog_pli_innovation_target_loc_value #script value that positions target marker
		}
		create_com_progress_bar_with_both = {
			progress_bar = mog_pli_proliferation_tracking_bar
			index = 1
			color = default
			bar_increase_color = green
			bar_decrease_color = bad
			drift_value = mog_pli_proliferation_tracking_bar_weekly_drift
			target_type = gold_bar
			target_value = mog_pli_proliferation_target_loc_value
		}
	}
	
	status_desc = {
	}

	on_weekly_pulse = {
		effect = {
			set_variable = {
				name = mog_pli_innovation_current_value
				value = "je:je_mog_pli_innovation.scripted_bar_progress(mog_pli_innovation_tracking_bar)"
			}
			set_variable = {
				name = mog_pli_proliferation_current_value
				value = "je:je_mog_pli_innovation.scripted_bar_progress(mog_pli_proliferation_tracking_bar)"
			}
			#set modifier from conditions
			MoG_PLI_do_innovation_stuff_effect = yes
		}
	}
}
