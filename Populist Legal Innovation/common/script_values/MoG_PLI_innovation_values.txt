### Innovation Limitations
mog_pli_max_innovation_potential_from_citizenship = {	# We use this for Loc and to apply a ceiling to any/all innovation incentive calculations.
	value = 1
	multiply = var:pli_innovation_citizenship_percent
	min = 0							# Should be strictly 0-1.
	max = 1							# But we'll clamp it to be safe.
}

mog_pli_max_proliferation_from_literacy = {	# We apply this as a ceiling to any/all proliferation calculations.
	value = 0
	add = {
		add = capital.literacy_rate		# Capital literacy_rate
		add = root.literacy_rate		# National literacy_rate
		divide = 2						# Get the average (should be value between 0-1)
	}
	multiply = 100						# Convert % into a whole number for the progress bar i.e. max proliferation rate is 100%
	max = 100							# But we'll clamp it to be safe.
}

mog_pli_max_proliferation_from_position = {	# We apply this as a ceiling to any/all proliferation calculations.
	if = {
		limit = { has_variable = pli_country_tech_percent_cache }
		value = var:pli_country_tech_percent_cache
	}
	else = {
		value = 100		# If it doesn't have the variable, don't accidentally crash the nation over it being missing.
	}
}
