REPLACE_OR_CREATE:law_mercantilism = {
	group = lawgroup_trade_policy
	
	icon = "gfx/interface/icons/law_icons/mercantilism.dds"
	
	progressiveness = 0

	is_visible = {
		OR = {
			AND = {
				com_law_total_seperation_trigger = yes
				# Any vanilla conditions that have been added
				always = yes
			}
			com_law_total_seperation_alternative_trigger = yes
		}
	}
	
	unlocking_technologies = {
		international_trade
	}

	modifier = {
		state_tariff_import_add = 0.50
		state_tariff_export_add = 0.20
		state_subvention_export_add = 0.50
		state_subvention_import_add = 0.20
		state_import_advantage_mult = -0.25
		state_export_advantage_mult = 0.125 # 0.25
		
		country_authority_mult = 0.1
		building_goods_input_mult = -0.05
		building_unincorporated_throughput_add = 0.05
		building_group_bg_manufacturing_throughput_add = 0.05
		building_group_bg_trade_allowed_collectivization_add = 0.5
		state_trade_capacity_mult = -0.2
	}
	
	# Don't change back to mercantilism once we have protectionism or free trade
	ai_will_do = {
		NOR = {
			has_law_or_variant = law_type:law_protectionism
			has_law_or_variant = law_type:law_free_trade
		}
	}
	
	ai_enact_weight_modifier = { #Petitions
		value = 0
		
		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
		
		if = {
			limit = {
				has_journal_entry = je_meiji_diplomacy
				has_law_or_variant = law_type:law_isolationism
			}
			add = 1000
		}
	}
}

REPLACE_OR_CREATE:law_free_trade = {
	group = lawgroup_trade_policy
	
	icon = "gfx/interface/icons/law_icons/free_trade.dds"
	
	progressiveness = 50

	is_visible = {
		OR = {
			AND = {
				com_law_total_seperation_trigger = yes
				# Any vanilla conditions that have been added
				always = yes
			}
			com_law_total_seperation_alternative_trigger = yes
		}
	}
	
	disallowing_laws = {
		law_command_economy
	}
	
	unlocking_technologies = {
		international_trade
	}
	
	on_enact = {
		if = {
			limit = { is_banning_goods = g:opium }
			remove_banned_goods = g:opium
		}
		if = {
			limit = { is_banning_goods = g:liquor }
			remove_banned_goods = g:liquor
		}
	}

	on_activate = {
		if = { # removes opium ban authority modifier if forced to open market via diplomacy
			limit = {
				AND = {
					c:CHI ?= this
					has_modifier = opium_ban_authority
				}
			}
			remove_modifier = opium_ban_authority
		}
	}
	
	modifier = {
		state_subvention_export_add = 0.50
		state_subvention_import_add = 0.50
		state_trade_advantage_mult = 0.25
		country_leverage_resistance_mult = -0.25
		
		state_trade_advantage_from_capacity_add = 0.0001	# On top of base_values = 0.0005
		state_max_trade_advantage_from_capacity_add = 0.2	# On top of base_values = 0.2
	}

	ai_enact_weight_modifier = { #Petitions
		value = 0
		
		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
		
		if = {
			limit = {
				has_journal_entry = je_meiji_diplomacy
				has_law_or_variant = law_type:law_isolationism
			}
			add = 1000
		}
	}
}

REPLACE_OR_CREATE:law_isolationism = {
	group = lawgroup_trade_policy
	
	icon = "gfx/interface/icons/law_icons/isolationism.dds"
	
	progressiveness = -50
	
	disallowing_laws = {
		law_laissez_faire
	}

	is_visible = {
		OR = {
			AND = {
				com_law_total_seperation_trigger = yes
				NOR = {
					AND = {
						any_primary_culture = {
							has_discrimination_trait = heritage_han
						}
						any_scope_state = {
							state_region = s:STATE_GUANGDONG
							is_largest_state_in_region = yes
						}
					}
					AND = {
						any_primary_culture = {
							cu:japanese = this
						}
						s:STATE_KYUSHU ?= {
							any_scope_state = {
								owner ?= ROOT
								is_largest_state_in_region = yes
							}
						}
					}
				}
			}
			com_law_total_seperation_alternative_trigger = yes
		}
	}
	
	modifier = {
		country_authority_mult = 0.5
		state_tax_capacity_mult = 0.25
		country_tech_spread_mult = -0.1 # -0.15
		
		country_leverage_resistance_mult = 1.0
		country_support_separatism_resistance_mult = 0.5
		
		# country_max_declared_interests_mult = -0.25 # -0.5
		
		country_disallow_trade_bool = yes
		
		building_goods_input_mult = -0.1
		building_throughput_add = 0.05
	}
	
	ai_will_do = {
		OR = {
			has_journal_entry = je_gyojo_shinwon
			ruler = {
				OR = {
					has_ideology = ideology:ideology_traditionalist
					has_ideology = ideology:ideology_vanguardist
				}
			}
		}
	}
	
	ai_enact_weight_modifier = { #Petitions
		value = 0
		
		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
	}
}

	icon = "gfx/interface/icons/law_icons/sakoku.dds"

	requires_law_or = {

	}

	is_visible = {
		OR = {
			AND = {
				com_law_sakoku_trigger = yes
				country_has_primary_culture = cu:japanese

					state_region = s:STATE_KYUSHU
					is_largest_state_in_region = yes
				}
			}
			com_law_sakoku_alternative_trigger = yes
		}
	}

		country_authority_mult = 0.25
		country_influence_mult = -0.50
		country_tech_spread_mult = -0.20
		state_trade_capacity_mult = -0.75
		state_tariff_import_add = 0.50
		state_tariff_export_add = 0.50
		state_subvention_export_add = 0.20
		state_subvention_import_add = 0.20
		country_leverage_resistance_mult = 0.75

		country_disallow_trade_outside_kyushu_bool = yes
		state_trade_center_max_limit_add = 5
		
		country_disallow_aggressive_plays_bool = yes
		
		building_goods_input_mult = -0.1
		building_throughput_add = 0.05
		state_max_trade_advantage_from_capacity_add = 0.3	# On top of base_values = 0.2 - only one state, high adv cap if big
		
		building_group_bg_trade_allowed_collectivization_add = 1
	}

	ai_will_do = {
		OR = {
			has_journal_entry = je_sakoku
			ruler = {
				OR = {
					has_ideology = ideology:ideology_traditionalist
					has_ideology = ideology:ideology_vanguardist
				}
			}
		}
	}

	on_deactivate = {
		if = {
			limit = {
				has_law = law_type:law_closed_borders
			}
			custom_tooltip = internal_controls_relaxed_tt
			activate_law = law_type:law_migration_controls
		}
		custom_tooltip = {
			text = landowners_stop_supporting_sakoku_tt
			ig:ig_landowners ?= {
				remove_ideology = ideology_japan_hierarchic
				add_ideology = ideology_japan_hierarchic_2
			}
		}
	}

	ai_enact_weight_modifier = { #Petitions
		value = 0

		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
	}
}
