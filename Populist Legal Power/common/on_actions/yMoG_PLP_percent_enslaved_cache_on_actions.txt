on_game_started = {
	on_actions = {
		culture_enslaved_percent_country_cache_on_game_started
	}
}

culture_enslaved_percent_country_cache_on_game_started = {
	effect = {
		every_country = {
			culture_enslaved_percent_country_cache_update = yes
		}
	}
}

	on_actions = {
		culture_enslaved_percent_country_cache_update_action
	}
}

culture_enslaved_percent_country_cache_update_action = {
	effect = {
		culture_enslaved_percent_country_cache_update = yes
	}
}
