			remove_global_variable = subject_initiator10
			clear_saved_scope = subject_initiator1
			clear_saved_scope = subject_initiator2
			clear_saved_scope = subject_initiator3
			clear_saved_scope = subject_initiator4
			clear_saved_scope = subject_initiator5
			clear_saved_scope = subject_initiator6
			clear_saved_scope = subject_initiator7
			clear_saved_scope = subject_initiator8
			clear_saved_scope = subject_initiator9
			clear_saved_scope = subject_initiator10
		}
		every_country = {
			limit = {
				has_variable = FX_potential_subject
			}
			remove_variable = FX_potential_subject
		}
	}
	
	option = {
		name = CANCEL
		default_option = yes
	}

	# conquer state
	option = {
		name = dp_conquer_state
		trigger = {
			scope:subject_initiator = { is_direct_subject_of = root }
			scope:FX_subject_fight_target_country = {
				NOT = { is_direct_subject_of = scope:subject_initiator }
				any_scope_state = {
					NOR = {
						has_claim_by = scope:subject_initiator
						is_treaty_port = yes
					}
				}
			}
		}
		trigger_event = {
			id = FX_subject_fight.11
			days = 0
			popup = yes
		}
	}
	
	# return state
	option = {
		name = dp_return_state
		trigger = {
			scope:subject_initiator = { is_direct_subject_of = root }
			scope:FX_subject_fight_target_country = {
				NOT = { is_direct_subject_of = scope:subject_initiator }
				any_scope_state = {
					OR = {
						has_claim_by = scope:subject_initiator
						# is_homeland_of_country_cultures = scope:subject_initiator
						# don't do this: the wargoal is invalid
					}
				}
			}
		}
		trigger_event = {
			id = FX_subject_fight.13
			days = 0
			popup = yes
		}
	}
	
	# take treaty port
	option = {
		name = dp_take_treaty_port
		trigger = {
			scope:subject_initiator = { is_direct_subject_of = root }
			scope:FX_subject_fight_target_country = {
				NOT = { is_direct_subject_of = scope:subject_initiator }
				any_scope_state = {
					is_treaty_port = yes
				}
			}
		}
		trigger_event = {
			id = FX_subject_fight.12
			days = 0
			popup = yes
		}
	}

	# annex country
	option = {
		name = dp_annex_war
		trigger = {
			scope:subject_initiator = { is_direct_subject_of = root }
			scope:FX_subject_fight_target_country = {
				OR = {
					any_scope_state = {
						count < 4
					}
					is_direct_subject_of = scope:subject_initiator
				}
			}
		}
		scope:subject_initiator = {
			create_diplomatic_play = {
				name = subject_fight_for_lord
				target_country = scope:FX_subject_fight_target_country
				type = dp_annex_war
			}
		}
	}
	
	# debug - abort and fix
	option = {
		name = FX_abort
		trigger = {
			scope:subject_initiator = { NOT = { is_direct_subject_of = root } }
		}
		scope:subject_initiator = {
			remove_variable = FX_potential_subject
		}
	}
}
