# Expects state scope
mog_spa_new_refresh_expected_sol_modifier = {
	random_scope_building = {
		limit = {
			OR = {
				is_subsistence_building = yes
				is_building_type = building_urban_center
			}
		}
		state = { save_temporary_scope_as = sol_calc_target_state }
	}
	scope:sol_calc_target_state = {
		if = {
			limit = { NOT = { exists = var:spa_state_locked } }
			#### TEMP REPLACEMENT
			if = {
				limit = { has_modifier = mog_lower_strata_expected_sol_from_average }
				remove_modifier = mog_lower_strata_expected_sol_from_average
			}
			if = {
				limit = { has_modifier = mog_middle_strata_expected_sol_from_average }
				remove_modifier = mog_middle_strata_expected_sol_from_average
			}
			if = {
				limit = { has_modifier = mog_upper_strata_expected_sol_from_average }
				remove_modifier = mog_upper_strata_expected_sol_from_average
			}
			
			if = {
				limit = { spa_sol_modification_trigger = yes }
				add_modifier = {
					name = mog_lower_strata_expected_sol_from_average
					multiplier = 1
				}
				add_modifier = {
					name = mog_middle_strata_expected_sol_from_average
					multiplier = 6
				}
				add_modifier = {
					name = mog_upper_strata_expected_sol_from_average
					multiplier = 11
				}
			}
			
			set_variable = {
				name = spa_state_locked
				value = 1
			}
		}
	}
}
