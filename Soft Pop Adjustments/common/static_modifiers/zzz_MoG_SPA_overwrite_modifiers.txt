# Add assimilation and education access penalties to states in turmoil
# Cannot use INJECT due to additive values which would be hugely broken if vanilla or some other mod changes the baseline.
# We use TRY_REPLACE in case something has completely redesigned these from the ground up.

# Applied if state is above 'in turmoil' threshold
# TRY_REPLACE:

	state_assimilation_mult = -0.3
	state_education_access_add = -0.3
